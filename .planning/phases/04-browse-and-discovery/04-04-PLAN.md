**Update `app/(app)/browse/page.tsx`:**

Add `getAllProperties()` call to fetch the full property list for recommendations:
```typescript
import { filterProperties, getAllProperties } from "@/lib/data"
import { BrowseClient } from "./_components/browse-client"

export default async function BrowsePage({ searchParams }: { searchParams: Promise<...> }) {
  const params = await searchParams
  // ...existing filter parsing...

  const allProperties = await getAllProperties()
  const properties = hasFilters
    ? await filterProperties({ category, region, minPrice, maxPrice })
    : allProperties

  // Apply BROWSE-14 limit when category filter active
  const displayProperties = category ? properties.slice(0, 8) : properties

  return (
    <BrowseClient
      properties={displayProperties}
      allProperties={allProperties}
      currentFilters={{ category, region, minPrice, maxPrice }}
      hasFilters={hasFilters}
    />
  )
}
```

Note: When no filters are active, `properties` and `allProperties` are the same query result — could optimize by reusing, but clarity is more important here.
