  **1. Create `app/(public)/p/[slug]/page.tsx`:**

  An async Server Component that fetches the property by slug and renders visible sections plus locked sections. Follow the exact same async params pattern as the private detail page at `app/(app)/listings/[slug]/page.tsx`.

  ```typescript
  import { notFound } from "next/navigation"
  import { getPropertyBySlug } from "@/lib/data"
  import { PropertyHero } from "@/components/property-detail/property-hero"
  import { PropertyMeta } from "@/components/property-detail/property-meta"
  import { PropertyAbout } from "@/components/property-detail/property-about"
  import { LockedSection } from "@/components/public-listing/locked-section"

  export default async function PublicListingPage({
    params,
  }: {
    params: Promise<{ slug: string }>
  }) {
    const { slug } = await params
    const property = await getPropertyBySlug(slug)

    if (!property) {
      notFound()
    }

    return (
      <div className="max-w-4xl mx-auto px-4 py-8">
        <PropertyMeta property={property} />
        <div className="mt-4">
          <PropertyHero images={[property.imageUrl]} />
        </div>
        <div className="mt-8 space-y-10">
          <PropertyAbout description={property.description} />
          <LockedSection title="Sponsorship Tiers" />
          <LockedSection title="Audience Demographics" />
          <LockedSection title="Activation Formats" />
        </div>
      </div>
    )
  }

  export async function generateStaticParams() {
    return []
  }
  ```

  Key differences from the private detail page:
  - `max-w-4xl` instead of `max-w-6xl` (single column, no sidebar)
  - No 5-col grid, no OfferSidebar
  - No PropertyDemographics, PropertyCategories, PropertyFormats â€” replaced with 3 LockedSection instances
  - No sidebar or app chrome (handled by the `(public)` layout)
